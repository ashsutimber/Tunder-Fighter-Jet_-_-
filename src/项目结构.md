\### 2.2 项目包结构

```
src/
├── Main.java                                    # 程序入口
├── Entity/                                      # 实体层
│   ├── Plane/
│   │   ├── Plane.java                          # 飞机抽象基类
│   │   ├── PlayerPlane.java                    # 玩家飞机
│   │   └── EnemyPlane.java                     # 敌机
│   ├── Bullet/
│   │   └── Bullet.java                         # 子弹抽象基类
│   ├── Skill/
│   │   └── Skill.java                          # 技能抽象基类
│   └── Buff/
│       └── Buff.java                           # 增益效果抽象基类
├── Controller/                                  # 控制层
│   ├── CoreController/
│   │   ├── GameEngine.java                     # 游戏主循环引擎
│   │   ├── GamePanel.java                      # 游戏界面 (JFrame)
│   │   ├── InputControl.java                   # 键盘输入处理
│   │   ├── EventControl.java                   # 事件监听/分发
│   │   └── GameEvent.java                      # 游戏事件定义
│   └── Coreiterface/
│       ├── PlayerActionListener.java           # 玩家操作监听接口
│       ├── EnemyActionListener.java            # 敌机行为监听接口
│       └── GameEventListener.java              # 游戏事件监听接口
├── image/                                       # 图片资源
│   ├── playerJet1.png                          # 玩家飞机1
│   ├── playerJet2.png                          # 玩家飞机2
│   ├── playerJet3.png                          # 玩家飞机3
│   ├── playerJet4.png                          # 玩家飞机4
│   ├── normalAnemyJet1.png                     # 普通敌机1
│   ├── normalAnemyJet2.png                     # 普通敌机2
│   ├── normalAnemyJet3.png                     # 普通敌机3
│   ├── normalAnemyJet4.png                     # 普通敌机4
│   ├── BossJet1.png                            # Boss敌机1
│   └── BossJet2.png                            # Boss敌机2

```
